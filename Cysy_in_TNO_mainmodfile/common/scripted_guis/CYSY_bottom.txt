scripted_gui = {
	
	CYSY_bottom_gui = {
		context_type = player_context
		parent_window_token = top_bar
		window_name = "CYSY_bottom_gui"
	
		visible = {
			original_tag = CYS
		}
		
		triggers = {
			chaoyangshixiao_crt_icon_visible = {
				original_tag = CYS
			}
		}
		effects = {
			chaoyangshixiao_crt_icon_click = {
				if = {
					limit = { NOT = { has_country_flag = CYSY_Map_GUI_open_flag } }
					set_country_flag = CYSY_Map_GUI_open_flag
					update_all_CYSY_control_pie_charts = yes
					update_CYSY_building_constructing_completed_left = yes
				}
				else_if = {
					limit = { has_country_flag = CYSY_Map_GUI_open_flag }
					clr_country_flag = CYSY_Map_GUI_open_flag
				}
			}
		}
	}
	CYSY_statistics_button = {
		context_type = player_context
		parent_window_token = top_bar
		window_name = "CYSY_top_statistics_bottom"
	
		visible = {
			original_tag = CYS
		}
		effects = {
			statistics_button_click = {
				if = {
					limit = { NOT = { has_country_flag = CYSY_statistics_GUI_open_flag } }
					set_country_flag = CYSY_statistics_GUI_open_flag
					set_variable = { CYSY_statistics_button_frame = 2}
				}
				else_if = {
					limit = { has_country_flag = CYSY_statistics_GUI_open_flag }
					clr_country_flag = CYSY_statistics_GUI_open_flag
					set_variable = { CYSY_statistics_button_frame = 1}
				}
			}
		}
		properties = {
			statistics_button = { frame = CYSY_statistics_button_frame }
		}
	}
	arrows_alert = {
		context_type = player_context
		parent_window_token = top_bar
		window_name = "arrows_alert"
	
		visible = {
			tag = CYS
			has_country_flag = CYSY_Map_GUI_open_flag
		}
	
		triggers = {				
			gui_alert_visible = {
				tag = CYS
			}
		}

		
	}
}
